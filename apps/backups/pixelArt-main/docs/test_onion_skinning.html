Onion Skinning Feature Test

This test demonstrates how onion skinning could work with the Pixel Art Studio. Onion skinning is a technique used in animation to see multiple frames at once, making it easier to create smooth animations.

How it works: When enabled, previous and next frames are displayed as semi-transparent overlays on the current frame, helping animators maintain consistency between frames.

Implementation Code Sample

Here's how the onion skinning feature could be implemented in the existing codebase:

// Add to state.js
State.onionSkinEnabled = true;
State.onionSkinOpacity = 0.5;
State.onionSkinFramesBefore = 1;
State.onionSkinFramesAfter = 1;
State.onionSkinColor = 'rgba(0, 128, 255, 0.3)';

// Add to canvas-manager.js
CanvasManager.renderWithOnionSkin() {
const ctx = UI.compositionCanvas.getContext('2d');
const currentFrame = State.frames[State.currentFrameIndex];

// Clear canvas
ctx.clearRect(0, 0, State.width, State.height);

// Draw onion skin frames (previous frames)
if (State.onionSkinEnabled) {
for (let i = 1; i <= State.onionSkinFramesBefore; i++) {
const prevFrameIndex = State.currentFrameIndex - i;
if (prevFrameIndex >= 0) {
const prevFrame = State.frames[prevFrameIndex];
this.drawFrameWithOpacity(prevFrame, State.onionSkinOpacity, 'rgba(0, 128, 255, 0.3)');
}
}
}

// Draw current frame normally
currentFrame.layers.forEach(layer => {
if (layer.visible) {
offCtx.clearRect(0, 0, State.width, State.height);
offCtx.putImageData(layer.data, 0, 0);
ctx.drawImage(State.offscreenCanvas, 0, 0);
}
});

// Draw onion skin frames (next frames)
if (State.onionSkinEnabled) {
for (let i = 1; i <= State.onionSkinFramesAfter; i++) {
const nextFrameIndex = State.currentFrameIndex + i;
if (nextFrameIndex < State.frames.length) {
const nextFrame = State.frames[nextFrameIndex];
this.drawFrameWithOpacity(nextFrame, State.onionSkinOpacity, 'rgba(255, 128, 0, 0.3)');
}
}
}

// Update preview
prevCtx.clearRect(0, 0, State.width, State.height);
prevCtx.drawImage(UI.compositionCanvas, 0, 0);
}

// Helper method
CanvasManager.drawFrameWithOpacity(frame, opacity, color) {
const ctx = UI.compositionCanvas.getContext('2d');

// Save current context state
ctx.save();

// Set global alpha for onion skin effect
ctx.globalAlpha = opacity;

// Draw each layer of the frame
frame.layers.forEach(layer => {
if (layer.visible) {
offCtx.clearRect(0, 0, State.width, State.height);
offCtx.putImageData(layer.data, 0, 0);

// Apply color tint
ctx.fillStyle = color;
ctx.fillRect(0, 0, State.width, State.height);
ctx.globalCompositeOperation = 'multiply';
ctx.drawImage(State.offscreenCanvas, 0, 0);
ctx.globalCompositeOperation = 'source-over';
}
});

// Restore context state
ctx.restore();
}
UI Integration

The onion skinning feature would integrate with the existing UI:

  • Timeline Controls: Add onion skin toggle button (already present in HTML)
  • Settings Panel: Add onion skin configuration options
  • Keyboard Shortcuts: Add Alt+O shortcut for quick toggling
  • Visual Feedback: Show onion skin indicators in the timeline
// Add to input-handler.js for keyboard shortcut
if (e.key === 'o' && e.altKey) {
e.preventDefault();
State.onionSkinEnabled = !State.onionSkinEnabled;
const onionBtn = document.getElementById('onionBtn');
if (onionBtn) {
onionBtn.classList.toggle('active', State.onionSkinEnabled);
}
CanvasManager.render();
}

// Add to UI for visual feedback
function updateOnionSkinUI() {
const onionBtn = document.getElementById('onionBtn');
if (onionBtn) {
onionBtn.classList.toggle('active', State.onionSkinEnabled);
onionBtn.title = State.onionSkinEnabled ? 'Disable Onion Skin (Alt+O)' : 'Enable Onion Skin (Alt+O)';
}

// Update timeline frame indicators
const frameElements = document.querySelectorAll('.frame-thumb');
frameElements.forEach((el, index) => {
el.classList.toggle('onion-before',
State.onionSkinEnabled &&
index > State.currentFrameIndex - State.onionSkinFramesBefore &&
index < State.currentFrameIndex
);
el.classList.toggle('onion-after',
State.onionSkinEnabled &&
index > State.currentFrameIndex &&
index <= State.currentFrameIndex + State.onionSkinFramesAfter
);
});
}
